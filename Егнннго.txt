#include "colors.inc"
#include "textures.inc"

camera { 
    location <15, 10, 25>   // Место, откуда вы смотрите на объекты 
    look_at <-9, -8, 1>     // и направление 
}

light_source { <-3, 10, -3> White shadowless } // Положение в пространстве и цвет

background { NeonBlue } // Неоново-голубой цвет неба

plane { 
    y, -10 // Плоскость в пространстве
    pigment { checker White, Red scale 3 } // Шахматная доска из белых и красных квадратов
}

// ========== СТУЛ С ЗАКРУГЛЕНИЯМИ ==========

// ---- ОСНОВНОЕ СИДЕНЬЕ ----

    <-8.5, -3.8, 5>,

    texture { pigment { color Gray } }

// ---- СПИНКА С ЗАКРУГЛЕНИЯМИ ----
union {
    // Основная спинка
    box { 
        <-8.5, -1, 5>, 
        <-8.3, 2, 10>
        texture { pigment { color Gray } }
    }
    
    // === ЗАКРУГЛЕНИЯ СПИНКИ (цилиндры) ===
    #local R = 0.08;  // радиус скругления
    
    // Верхние углы спинки
    cylinder { <-8.3, 2, 5>, <-8.3, 2, 5.1>, R }
    cylinder { <-8.3, 2, 9.9>, <-8.3, 2, 10>, R }
    
    // Нижние углы спинки
    cylinder { <-8.5, -1, 5>, <-8.5, -1, 5.1>, R }
    cylinder { <-8.5, -1, 9.9>, <-8.5, -1, 10>, R }
    
    // Сферы для идеального скругления углов
    sphere { <-8.4, 2, 5.05>, R }
    sphere { <-8.4, 2, 9.95>, R }
    sphere { <-8.6, -1, 5.05>, R }
    sphere { <-8.6, -1, 9.95>, R }
    
    // Скругления на верхней грани
    cylinder { <-8.5, 2, 5.1>, <-8.5, 2, 9.9>, R }
    
    texture { pigment { color Gray } }
}

// ---- ЗАКРУГЛЕНИЯ НА СИДЕНЬЕ ----

    // Скругления передних углов сиденья

    texture { pigment { color Gray } }
